    	send(key, message) {
		if (!this._eventKeyExists(key)) {
			throw `key '${key}' does not exist in EventKeys`;
		}

		return this.eventRegistry.trigger(key, message);
	}

	_eventKeyExists(key) {
		const keys = [];
		for (let property in EventKeys) {
			let value = EventKeys[property];
			keys.push(value);
		}
		return keys.indexOf(key) >= 0;
	}
